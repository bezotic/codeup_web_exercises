Function- is a reusable block of code - reusable block of code with a name - lil' robot ( take in data and do stuff to it, return data) Why - saves time - convenient b/c we call the function to run later. - more readable code - accessibility - modularity where - function definitions at the top of our code - function calls anywhere we need the return value #Anatomy of a function definition function sayGoodMorning(name) { return "Good morning"; } #Calling or running a function var message = sayGoodMorning('Richard'); console.log(message); message = sayGoodMorning('Bob'); console.log(message); Return Values function square(n) { if(isNaN(n)) { console.log(n + ' is not a number'); return false; } return n * n; } function sumOfSquares(a,b) { return square(a) + square(b); } function differenceOfSquares(a,b) { return square(a) - square(b); }
